<objective>
Add POST /v1/orders/voice — an internal endpoint that lets n8n create orders on behalf of a buyer without requiring the buyer's JWT. Protected by AI_SERVER_SHARED_SECRET + userId in the request body. Also configure n8n's Supabase credential so it can read/write session_memory.

```typescript
function isValidSharedSecret(providedSecret: string) {
  if (!env.AI_SERVER_SHARED_SECRET) return false;
  const providedBuffer = Buffer.from(providedSecret, "utf8");
  const expectedBuffer = Buffer.from(env.AI_SERVER_SHARED_SECRET, "utf8");
  return (
    providedSecret.length > 0 &&
    providedBuffer.length === expectedBuffer.length &&
    crypto.timingSafeEqual(providedBuffer, expectedBuffer)
  );
}
```

```typescript
voiceOrderRouter.post(
  "/voice",
  // Step 1: Authenticate via shared secret and parse body to get userId
  (req, res, next) => {
    if (!env.AI_SERVER_SHARED_SECRET) {
      return res.status(503).json({ error: { code: "AI_SERVER_SHARED_SECRET_MISSING", message: "Server misconfiguration" } });
    }
    const provided = String(req.headers["x-ai-server-secret"] ?? "");
    if (!isValidSharedSecret(provided)) {
      return res.status(401).json({ error: { code: "UNAUTHORIZED", message: "Invalid AI server shared secret" } });
    }
    // Parse body early to extract userId for idempotency hash
    const parsed = VoiceCreateOrderSchema.safeParse(req.body);
    if (!parsed.success) {
      return res.status(400).json({ error: { code: "VALIDATION_ERROR", details: parsed.error.flatten() } });
    }
    // Patch req.auth BEFORE idempotency middleware runs so the hash includes userId
    req.auth = { userId: parsed.data.userId, sessionId: undefined, realm: "app", role: "buyer" };
    req.body = parsed.data; // replace with typed+validated body
    return next();
  },
  requireIdempotency({ scope: "order_create" }),
  async (req, res) => {
    const input = req.body as z.infer<typeof VoiceCreateOrderSchema>;
    // Full transaction body identical to POST "/" — copy verbatim, replace req.auth!.userId with input.userId
    const client = await pool.connect();
    try {
      await client.query("BEGIN");
      // ... [copy lot validation, order insert, order_items insert, order_events insert, enqueueOutboxEvent, COMMIT from POST /]
    } catch {
      await client.query("ROLLBACK");
      return res.status(500).json({ error: { code: "INTERNAL_ERROR", message: "Voice order create failed" } });
    } finally {
      client.release();
    }
  }
```
