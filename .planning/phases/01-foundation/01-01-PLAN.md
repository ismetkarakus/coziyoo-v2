   ```bash

   ```

2. Update `apps/voice-dashboard/package.json`:
   - Set `"name": "@coziyoo/voice-dashboard"`
   - Set `"private": true`
   - Set `"version": "0.1.0"`
   - Change dev script to: `"dev": "next dev --turbopack --port 3001"`
   - Add: `"typecheck": "tsc --noEmit"`
   - Keep build script as: `"build": "next build"`

3. Update `apps/voice-dashboard/next.config.ts` to:
   ```typescript
   import type { NextConfig } from "next";

   const nextConfig: NextConfig = {
     output: "standalone",
     transpilePackages: ["@coziyoo/shared-types", "@coziyoo/shared-utils"],
     async rewrites() {
       return [
         {
           source: "/v1/:path*",
           destination: "http://localhost:3000/v1/:path*",
         },
       ];
     },
   };

   export default nextConfig;
   ```

4. Create `apps/voice-dashboard/.env.local` with:
   ```
   NEXT_PUBLIC_API_BASE_URL=
   ```
   Note: Empty string means requests go to same origin, which the dev rewrites proxy to localhost:3000. In production, this will be set to `https://api.coziyoo.com`.

5. Update root `package.json`:
   - Add `"apps/voice-dashboard"` to the `workspaces` array (after `"apps/admin"`)
   - Add script: `"dev:voice-dashboard": "npm run dev --workspace=apps/voice-dashboard"`
   - Add script: `"build:voice-dashboard": "npm run build --workspace=apps/voice-dashboard"`

6. Create `apps/voice-dashboard/src/lib/types.ts` with the AdminUser, Tokens, and ApiError types (ported from `apps/admin/src/types/core.ts` but only the three types needed -- NOT Language, Dictionary, or GlobalSearch types):
